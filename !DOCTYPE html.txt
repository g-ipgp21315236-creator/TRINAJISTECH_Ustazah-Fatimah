    const AudioContext = window.AudioContext || window.webkitAudioContext;
    let audioCtx;
    
    function initAudio() {
        if (!audioCtx) audioCtx = new AudioContext();
        if (audioCtx.state === 'suspended') audioCtx.resume();

    function playTone(freq, type, duration, vol=0.1) {
        initAudio();
        const osc = audioCtx.createOscillator();
        const gain = audioCtx.createGain();
        osc.type = type;
        osc.frequency.setValueAtTime(freq, audioCtx.currentTime);
        gain.gain.setValueAtTime(vol, audioCtx.currentTime);
        gain.gain.exponentialRampToValueAtTime(0.001, audioCtx.currentTime + duration);
        osc.connect(gain);
        gain.connect(audioCtx.destination);
        osc.start();
        osc.stop(audioCtx.currentTime + duration);
    }

    function createBoard() {
        const board = document.getElementById('game-board');
        board.innerHTML = '';
        for (let r = 9; r >= 0; r--) {
            for (let c = 0; c < 10; c++) {
                let col = (r % 2 === 0) ? c : (9 - c);
                let num = (r * 10) + col + 1;
                const cell = document.createElement('div');
                cell.className = `cell ${num === 100 ? 'bg-win' : `bg-${(num % 5) + 1}`}`;
                cell.id = `cell-${num}`;
                
                const label = document.createElement('span');
                label.innerText = num;
                label.className = "absolute top-1 left-2 text-[10px] opacity-20 pointer-events-none";
                if(num === 100) label.style.opacity = "0"; 
                cell.appendChild(label);

                if (num === 100) {
                    const winContent = document.createElement('div');
                    winContent.innerHTML = '<div style="font-size:2.5rem;" class="pointer-events-none">🏆</div>';
                    cell.appendChild(winContent);
                } else if (gifts[num]) {
                    const colorMap = {easy:'bg-green-400', medium:'bg-blue-400', hard:'bg-red-400'};
                    cell.innerHTML += `<div class="gift-box ${colorMap[gifts[num]]} pointer-events-none">🎁</div>`;
                }
                board.appendChild(cell);
            }
        }
        initTokens();
        drawOverlays();
        updateTurnDisplay();
        renderEditorList();
    }

    function initTokens() {
        const container = document.getElementById('tokens-container');
        container.innerHTML = '';
        players.forEach(p => {
            const t = document.createElement('div');
            t.id = `token-${p.id}`;
            t.className = 'player-token pointer-events-none';
            t.style.background = p.color;
            t.setAttribute('data-emoji', p.emoji);
            container.appendChild(t);
        });
        positionTokens();
    }

    function positionTokens() {
        const board = document.getElementById('game-board');
        const boardRect = board.getBoundingClientRect();
        players.forEach(p => {
            const cell = document.getElementById(`cell-${p.pos}`);
            const cellRect = cell.getBoundingClientRect();
            const t = document.getElementById(`token-${p.id}`);
            
            // Calculate center of cell relative to board container
            const x = (cellRect.left - boardRect.left) + (cellRect.width / 2);
            const y = (cellRect.top - boardRect.top) + (cellRect.height / 2);
            
            t.style.left = `${x}px`;
            t.style.top = `${y}px`;
        });
    }

    function getPos(num) {
        const cell = document.getElementById(`cell-${num}`);
        const board = document.getElementById('game-board');
        const rect = cell.getBoundingClientRect();
        const bRect = board.getBoundingClientRect();
        return {
            x: (rect.left - bRect.left + rect.width/2) * (1000/bRect.width),
            y: (rect.top - bRect.top + rect.height/2) * (1000/bRect.height)
        };
    }

    function drawOverlays() {
        const svg = document.getElementById('overlay');
        svg.innerHTML = '';
        ladders.forEach(l => {
            const p1 = getPos(l.start), p2 = getPos(l.end);
            const dx = p2.x - p1.x, dy = p2.y - p1.y;
            const dist = Math.sqrt(dx*dx + dy*dy);
            const angle = Math.atan2(dy, dx);
            const offset = 14; 

            const ox = Math.cos(angle + Math.PI/2) * offset;
            const oy = Math.sin(angle + Math.PI/2) * offset;

            const line = (x1, y1, x2, y2, cls) => {
                const el = document.createElementNS("http://www.w3.org/2000/svg", "line");
                el.setAttribute("x1", x1); el.setAttribute("y1", y1); el.setAttribute("x2", x2); el.setAttribute("y2", y2);
                el.setAttribute("class", cls); svg.appendChild(el);
            };
            
            line(p1.x + ox, p1.y + oy, p2.x + ox, p2.y + oy, "ladder-rail");
            line(p1.x - ox, p1.y - oy, p2.x - ox, p2.y - oy, "ladder-rail");
            
            const numSteps = Math.floor(dist / 25);
            for(let i=1; i <= numSteps; i++) {
                const r = i / (numSteps + 1);
                const stepX = p1.x + dx * r;
                const stepY = p1.y + dy * r;
                const stepOx = Math.cos(angle + Math.PI/2) * (offset + 3);
                const stepOy = Math.sin(angle + Math.PI/2) * (offset + 3);
                line(stepX + stepOx, stepY + stepOy, stepX - stepOx, stepY - stepOy, "ladder-step");
            }
        });

        snakes.forEach((s, i) => {
            const h = getPos(s.start), t = getPos(s.end);
            const dx = t.x - h.x;
            const dy = t.y - h.y;
            const angle = Math.atan2(dy, dx);
            
            const cp1x = h.x + dx * 0.3 + Math.cos(angle + Math.PI/2) * (i%2?70:-70);
            const cp1y = h.y + dy * 0.3 + Math.sin(angle + Math.PI/2) * (i%2?70:-70);
            const cp2x = h.x + dx * 0.7 + Math.cos(angle - Math.PI/2) * (i%2?70:-70);
            const cp2y = h.y + dy * 0.7 + Math.sin(angle - Math.PI/2) * (i%2?70:-70);

            const d = `M ${h.x} ${h.y} C ${cp1x} ${cp1y}, ${cp2x} ${cp2y}, ${t.x} ${t.y}`;
            
            const g = document.createElementNS("http://www.w3.org/2000/svg", "g");

            const glow = document.createElementNS("http://www.w3.org/2000/svg", "path");
            glow.setAttribute("d", d); glow.setAttribute("class", "snake-glow"); glow.setAttribute("stroke", s.glow);
            g.appendChild(glow);

            const b = document.createElementNS("http://www.w3.org/2000/svg", "path");
            b.setAttribute("d", d); b.setAttribute("class", "snake-body"); b.setAttribute("stroke", s.color);
            g.appendChild(b);

            const pattern = document.createElementNS("http://www.w3.org/2000/svg", "path");
            pattern.setAttribute("d", d); pattern.setAttribute("stroke", "rgba(255,255,255,0.4)");
            pattern.setAttribute("stroke-width", "6"); pattern.setAttribute("fill", "none");
            pattern.setAttribute("stroke-dasharray", "10,15"); pattern.setAttribute("stroke-linecap", "round");
            g.appendChild(pattern);

            const angleDeg = (angle * 180 / Math.PI) + 90;
            const headGroup = document.createElementNS("http://www.w3.org/2000/svg", "g");
            headGroup.setAttribute("transform", `translate(${h.x}, ${h.y}) rotate(${angleDeg})`);

            const snakeHead = document.createElementNS("http://www.w3.org/2000/svg", "ellipse");
            snakeHead.setAttribute("cx", "0"); snakeHead.setAttribute("cy", "0");
            snakeHead.setAttribute("rx", "14"); snakeHead.setAttribute("ry", "10");
            snakeHead.setAttribute("fill", s.color);
            headGroup.appendChild(snakeHead);

            const eye1 = document.createElementNS("http://www.w3.org/2000/svg", "circle");
            eye1.setAttribute("cx", "-5"); eye1.setAttribute("cy", "-3"); eye1.setAttribute("r", "3.5"); eye1.setAttribute("fill", "white");
            const eye2 = document.createElementNS("http://www.w3.org/2000/svg", "circle");
            eye2.setAttribute("cx", "5"); eye2.setAttribute("cy", "-3"); eye2.setAttribute("r", "3.5"); eye2.setAttribute("fill", "white");
            headGroup.appendChild(eye1); headGroup.appendChild(eye2);

            const tongue = document.createElementNS("http://www.w3.org/2000/svg", "path");
            tongue.setAttribute("d", `M 0 -10 L -4 -20 M 0 -10 L 4 -20`);
            tongue.setAttribute("class", "snake-tongue");
            headGroup.appendChild(tongue);

            g.appendChild(headGroup);
            svg.appendChild(g);
        });
    }

    async function handleRoll() {
        if(isRolling) return;
        initAudio();
        isRolling = true;
        document.getElementById('roll-btn').disabled = true;
        
        const diceDisplay = document.getElementById('dice-display');
        diceDisplay.classList.add('rolling-dice');
        
        let roll = 0;
        const dice = ['⚀','⚁','⚂','⚃','⚄','⚅'];
        for(let i=0; i<12; i++) {
            roll = Math.floor(Math.random()*6)+1;
            diceDisplay.innerText = dice[roll-1];
            soundRoll();
            await new Promise(r => setTimeout(r, 60));
        }
        diceDisplay.classList.remove('rolling-dice');

        let p = players[turn];
        for(let i=0; i<roll; i++) {
            if(p.pos < 100) { 
                p.pos++; 
                positionTokens(); 
                soundStep();
                
                const currentCell = document.getElementById(`cell-${p.pos}`);
                currentCell.style.transform = 'scale(1.2)';
                setTimeout(() => currentCell.style.transform = '', 150);
                await new Promise(r => setTimeout(r, 200)); 
            }
        }
        checkPosition(p);
    }

    function checkPosition(p) {
        const ladder = ladders.find(l => l.start === p.pos);
        const snake = snakes.find(s => s.start === p.pos);
        const diff = gifts[p.pos];
        if(diff) {
            setTimeout(() => showQuestion(p, diff), 400);
        } else if(ladder) { 
            setTimeout(() => { 
                p.pos = ladder.end; 
                positionTokens(); 
                soundLadder(); 
                nextTurn(); 
            }, 500);
        } else if(snake) { 
            setTimeout(() => { 
                p.pos = snake.end; 
                positionTokens(); 
                soundSnake(); 
                nextTurn(); 
            }, 500);
        } else {
            nextTurn();
        }
    }

    function showQuestion(p, diff) {
        const pool = questionsbank.filter(q => q.diff === diff);
        const q = pool.length > 0 ? pool[Math.floor(Math.random()*pool.length)] : questionsbank[0];
        
        const modal = document.getElementById('question-modal');
        modal.style.display = 'flex';

        document.getElementById('q-text').innerText = q.text;
        const cont = document.getElementById('options-container');
        cont.innerHTML = '';
        const correctVal = q.options[q.correct];
        
        const colors = {easy:'bg-green-500', medium:'bg-blue-500', hard:'bg-red-500'};
        const tag = document.getElementById('q-category-tag');
        tag.className = `absolute -top-6 left-1/2 -translate-x-1/2 px-8 py-2 rounded-full text-white font-black text-sm shadow-lg ${colors[diff]}`;
        tag.innerText = diff.toUpperCase();

        const shuffledOptions = [...q.options].sort(()=>Math.random()-0.5);
        shuffledOptions.forEach(opt => {
            const b = document.createElement('button');
            b.className = "p-4 bg-slate-50 border-2 rounded-2xl font-bold hover:border-yellow-400 hover:bg-yellow-50 transition-all text-slate-700";
            b.innerText = opt;
            b.onclick = async () => {
                initAudio();
                modal.style.display = 'none';
                
                if(opt === correctVal) { 
                    p.score += (diff==='easy'?10:diff==='medium'?20:30); 
                    soundCorrect();
                    confetti({particleCount: 150, spread: 80, origin: { y: 0.6 }}); 
                    nextTurn();
                } else { 
                    soundWrong();
                    await new Promise(r => setTimeout(r, 400));
                    
                    let penalty = diff === 'easy' ? 1 : (diff === 'medium' ? 3 : 5);
                    let target = Math.max(1, p.pos - penalty);
                    
                    while(p.pos > target) {
                        p.pos--;
                        positionTokens();
                        soundStep();
                        await new Promise(r => setTimeout(r, 200));
                    }
                    nextTurn();
                }
            };
            cont.appendChild(b);
        });
    }

    function nextTurn() {
        if(players[turn].pos >= 100) {
            document.getElementById('game-status').innerHTML = `<span class="text-green-600 font-black">PEMAIN ${players[turn].emoji} MENANG!</span>`;
            soundCorrect();
            setTimeout(soundCorrect, 200);
            confetti({particleCount:500, spread:160});
            return;
        }
        turn = (turn + 1) % players.length;
        isRolling = false;
        document.getElementById('roll-btn').disabled = false;
        updateTurnDisplay();
    }

    function updateTurnDisplay() {
        const tracker = document.getElementById('turn-tracker');
        tracker.innerHTML = '';
        players.forEach((p, i) => {
            const active = i === turn;
            const card = document.createElement('div');
            card.className = `p-2 md:p-3 rounded-xl bg-white border-2 text-center transition-all duration-300 ${active?'border-green-500 shadow-xl scale-110 active-turn -translate-y-1': 'border-slate-100 opacity-60'}`;
            card.innerHTML = `<div class="text-xl md:text-2xl">${p.emoji}</div><div class="text-[10px] font-black">${p.score}</div>`;
            tracker.appendChild(card);
        });
        document.getElementById('game-status').innerText = `Giliran: ${players[turn].emoji}`;
    }

    function addNewQuestion() {
        const text = document.getElementById('new-q-text').value;
        const opt0 = document.getElementById('new-q-opt0').value;
        if(!text || !opt0) return;
        
        questionsbank.unshift({
            id: Date.now(), 
            text, 
            options: [
                opt0, 
                document.getElementById('new-q-opt1').value || "Salah", 
                document.getElementById('new-q-opt2').value || "Salah", 
                document.getElementById('new-q-opt3').value || "Salah"
            ],
            correct: 0, 
            diff: document.getElementById('new-q-diff').value
        });
        
        renderEditorList();
        // Clear inputs
        document.getElementById('new-q-text').value = '';
        document.getElementById('new-q-opt0').value = '';
        document.getElementById('new-q-opt1').value = '';
        document.getElementById('new-q-opt2').value = '';
        document.getElementById('new-q-opt3').value = '';
    }

    function deleteQuestion(id) {
        questionsbank = questionsbank.filter(q => q.id !== id);
        renderEditorList();
    }
